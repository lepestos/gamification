base_url: "localhost:8000/api/v1"

/product
    GET: отобразить список продуктов
    POST: добавить продукт
        body: {
            "name": String,
            "price": Decimal
        }

/product/:id
    GET: отобразить продукт
    PUT/PATCH: изменить продукт
    DELETE: удалить продукт

/black-box
    GET: отобразить коробку
    POST: добавить коробку
        body: {
            "name": String,
            "products": [id_1, id_2, id_3],
            "probabilities": [prob_1, prob_2, prob_3]
        }

/black-box/:id
    GET: отобразить коробку
    PUT/PATCH: изменить коробку
        body: {
            "name": String,
            "products": [id_1, id_2, id_3],
            "probabilities": [prob_1, prob_2, prob_3]
        }
    DELETE: удалить коробку

/black-box/calculate
    POST: расчитать оптимальную модель коробки
        body: {
            "prices": [pr_1, pr_2, pr_3],
            "max_count_costly": Integer,
            "profit": Decimal, // десятичная дробь, не проценты!
            "loyalty": Decimal // десятичная дробь, не проценты!
        }
        response: {
            probabilities: [prob_1, prob_2, prob_3],
            amounts: [a_1, a_2, a_3]
        }
